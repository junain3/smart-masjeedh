const normalizeRelationship = (value?: string) => {
  const map: Record<string, string> = {
    "Head": "Family Head",
    "Wife": "Wife", 
    "Son": "Son",
    "Daughter": "Daughter",
    "Father": "Father",
    "Mother": "Mother",
    "Other": "Other",
  };
  return map[value || ""] || value || "-";
};
